// Type of unicode [normalization](http://www.unicode.org/reports/tr15)
pub type normalizationstd/text/unicode/normalization: V {
  // Canonical composition (default). Use composed characters when possible.
  con NFCstd/text/unicode/NFC: normalization
  // Canonical decomposition.
  con NFDstd/text/unicode/NFD: normalization
  // Compatibility composition.
  con NFKCstd/text/unicode/NFKC: normalization
  // Compatibility decomposition.
  con NFKDstd/text/unicode/NFKD: normalization
}
